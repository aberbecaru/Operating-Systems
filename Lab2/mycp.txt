#include<unistd.h>

#include<stdio.h>

#include<stdlib.h>

#include<fcntl.h>

#include<errno.h>

int main(int argc, char* argv[]){

int f1, f2;

char buffer[2048];



if(argc < 3){

	perror("Prea putine argumente");

	return errno;

}





f1 = open(argv[1], O_RDONLY);

if(f1 == -1){

	perror("Primul fisier nu s-a deschis");

	return errno;

}

	

f2 = open(argv[2], O_WRONLY);

if(f2 == -1){

	perror("Al doilea fisier nu s-a deschis");

	return errno;

}



while(read(f1, &buffer, 1) != 0)

	write(f2, &buffer, 1);

	



return 0;

}
